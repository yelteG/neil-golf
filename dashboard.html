""" local_server.py --------------- Run this in the background: python local_server.py It listens on http://localhost:5000 for data from the Chrome extension and pushes new rounds to Google Sheets using neil_to_sheets.py logic. """ import json import os import pickle import sys from http.server import BaseHTTPRequestHandler, HTTPServer from datetime import datetime # ── Config ──────────────────────────────────────────────────────────────────── SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__)) SECRETS_FILE = os.path.join(SCRIPT_DIR, 'client_secrets.json') TOKEN_FILE = os.path.join(SCRIPT_DIR, 'oauth_token.pkl') SHEET_ID = '1hySN7Csgydi_zg1sNGf7Z-3-Z6PS4J-zzTef8uk26QQ' TAB_NAME = 'Neil Rounds' PORT = 5000 HEADERS = ['Date','Course','Tee','Score','Scratch Rating','Slope', 'Daily Diff','Daily Hcp','GA Handicap','Gross Diff','Prior GA Hcp','Home Club'] # ── Google Sheets helpers ───────────────────────────────────────────────────── def get_sheets_service(): from google.oauth2.credentials import Credentials from google_auth_oauthlib.flow import InstalledAppFlow from google.auth.transport.requests import Request from googleapiclient.discovery import build SCOPES = ['https://www.googleapis.com/auth/spreadsheets'] creds = None if os.path.exists(TOKEN_FILE): with open(TOKEN_FILE, 'rb') as f: creds = pickle.load(f) if not creds or not creds.valid: if creds and creds.expired and creds.refresh_token: creds.refresh(Request()) else: flow = InstalledAppFlow.from_client_secrets_file(SECRETS_FILE, SCOPES) creds = flow.run_local_server(port=0) with open(TOKEN_FILE, 'wb') as f: pickle.dump(creds, f) return build('sheets', 'v4', credentials=creds) def get_existing_rows(service): result = service.spreadsheets().values().get( spreadsheetId=SHEET_ID, range=f'{TAB_NAME}!A:L' ).execute() return result.get('values', []) def push_to_sheets(new_rows): service = get_sheets_service() existing = get_existing_rows(service) # Build dedup set from existing rows (skip header) existing_keys = set() for row in existing[1:]: if len(row) >= 4: key = f"{row[0]}|{row[1]}|{row[3]}" # Date|Course|Score existing_keys.add(key) # Find rows to add to_add = [] for r in new_rows: key = f"{r.get('Date','')}|{r.get('Course','')}|{r.get('Score','')}" if key not in existing_keys: to_add.append([str(r.get(h, '')) for h in HEADERS]) if not to_add: return 0, len(existing) - 1 # Write header if sheet is empty if len(existing) == 0: service.spreadsheets().values().update( spreadsheetId=SHEET_ID, range=f'{TAB_NAME}!A1', valueInputOption='USER_ENTERED', body={'values': [HEADERS]} ).execute() # Append new rows - use USER_ENTERED so dates are interpreted correctly service.spreadsheets().values().append( spreadsheetId=SHEET_ID, range=f'{TAB_NAME}!A1', valueInputOption='USER_ENTERED', insertDataOption='INSERT_ROWS', body={'values': to_add} ).execute() total = len(existing) - 1 + len(to_add) return len(to_add), total # ── HTTP Server ─────────────────────────────────────────────────────────────── class Handler(BaseHTTPRequestHandler): def do_OPTIONS(self): # Handle CORS preflight from Chrome extension self.send_response(200) self._cors_headers() self.end_headers() def do_POST(self): if self.path != '/update': self.send_response(404) self.end_headers() return length = int(self.headers.get('Content-Length', 0)) body = self.rfile.read(length) try: data = json.loads(body) rows = data.get('rows', []) print(f"\n[{datetime.now().strftime('%H:%M:%S')}] Received {len(rows)} rows from extension") added, total = push_to_sheets(rows) print(f" ✅ Added {added} new rows | Total: {total} rows in sheet") self.send_response(200) self._cors_headers() self.send_header('Content-Type', 'application/json') self.end_headers() self.wfile.write(json.dumps({'added': added, 'total': total}).encode()) except Exception as e: print(f" ❌ Error: {e}") self.send_response(500) self._cors_headers() self.send_header('Content-Type', 'application/json') self.end_headers() self.wfile.write(json.dumps({'error': str(e)}).encode()) def _cors_headers(self): self.send_header('Access-Control-Allow-Origin', '*') self.send_header('Access-Control-Allow-Methods', 'POST, OPTIONS') self.send_header('Access-Control-Allow-Headers', 'Content-Type') def log_message(self, format, *args): pass # Suppress default HTTP logs # ── Main ────────────────────────────────────────────────────────────────────── if __name__ == '__main__': print("=" * 50) print(" Neil's Golf Scraper — Local Server") print(f" Listening on http://localhost:{PORT}") print(" Keep this running while using the extension") print("=" * 50) server = HTTPServer(('localhost', PORT), Handler) try: server.serve_forever() except KeyboardInterrupt: print("\nServer stopped.")
